In[162]:
def codo(rango):
    sse = []
    k_range = list(range(1, rango))

    for k in k_range:
        km = KMeans(n_clusters=k, n_init='auto', random_state=1004)
        km.fit(X)
        sse.append(km.inertia_)

    plt.plot(k_range, sse, '-or')
    plt.xlabel("k")
    plt.ylabel('Suma de errores cuadrados')
    plt.show()
